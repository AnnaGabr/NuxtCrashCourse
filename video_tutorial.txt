> To add bootstrap:
in nuxt.config.ts:
export default defineNuxtConfig({
    meta:{
        link: [
            {
                rel: "stylesheet",
                href: "https://cdn.jsdelivr.net/npm/bootstrap@5.2.0/dist/css/bootstrap.min.css",
            }
        ]
    }
})
